import pandas as pd
import random
from sklearn.linear_model import LinearRegression

# create dataset
fish_species = ['Tuna', 'Salmon', 'Trout', 'Bass', 'Sardine', 'Cod', 'Mackerel']

weights = []

for i in range(50):
    fish_weight = []
    for j in range(7):
        weight = random.randint(1, 20)
        fish_weight.append(weight)
    weights.append(fish_weight)

df = pd.DataFrame(weights, columns=fish_species)

# independent variables (all except last column)
X = df.iloc[:, :-1]

# target variable (last column)
y = df.iloc[:, -1]

# model
model = LinearRegression()
model.fit(X, y)

# predict new fish
new_fish = [[10, 12, 15, 7, 4, 8]]  # 6 inputs (because last column is target)

predicted_weight = model.predict(new_fish)

print("Predicted weight:", predicted_weight)
